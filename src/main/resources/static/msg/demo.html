数据结构 栈应用 1 括号匹配

大概是一道栈的应用的水题吧。
输入一行符号,以#结束，判断其中的括号是否匹配。
遇到左括号入栈，遇到右括号出栈。如果字符串结束栈不空，输出栈内左括号对应右括号。

Question

用栈实现：输入一行符号,以#结束，判断其中的括号是否匹配。括号包括：

  1. { } 、 [ ] 、 ( )、  < >

如果匹配，输出 right
如果不匹配，给出错误提示。包括：
第几个符号处理时出现错误；哪几个符号失配等

Example





















思路

遇到左括号入栈，遇到右括号出栈。如果字符串结束栈不空，输出栈内左括号对应右括号。

没有一遍过，因为题目中没说如果出错就不再继续处理。

相关

  1. 栈：顺序表，只允许在一端输入输出。
  2. getline(cin,str);可输入整行，支持空格。
  3. 如果用数组实现栈，则其存储空间必须是一次性申请得到的。

Answer

指定初始maxsize=1是我某种程度上的强迫症，反正这题时间限制很宽松。

  1. #include <iostream>
  2. #include<cstring>
  3. #include<string>
  5. using namespace std;
  7. class mystack{
  8. private:
  9.     char *data;
  10.     int maxsize;
  11.     int top;
  12. public:
  13.     mystack(){
  14.         maxsize=1;
  15.         data=new char[maxsize];
  16.         top=-1;
  17.     }
  19.     bool isEmpty(){
  20.         return top==-1;
  21.     }
  23.     char getTop(){
  24.         if(top>-1){
  25.             return data[top];
  26.         }
  27.         return '\0';
  28.     }
  30.     void pop(){
  31.         if(top>=0){
  32.             top--;
  33.         }
  34.     }
  36.     void expand(){
  37.         char*tmp=new char[maxsize*2];
  38.         strncpy(tmp,data,maxsize);
  39.         maxsize*=2;
  40.         delete[] data;
  41.         data=tmp;
  42.     }
  44.     void push(char x){
  45.         if(top+1==maxsize){
  46.             expand();
  47.         }
  48.         data[++top]=x;
  49.     }
  51.     ~mystack(){
  52.         delete[] data;
  53.     }
  54. };
  56. char getPar(char x){
  57.     switch(x){
  58.     case '{':
  59.         return '}';
  60.     case '[':
  61.         return ']';
  62.     case '(':
  63.         return ')';
  64.     case '<':
  65.         return '>';
  66.     }
  67.     return '\0';
  68. }
  70. int isKH(char x){
  71.     switch(x){
  72.     case '{':
  73.     case '[':
  74.     case '(':
  75.     case '<':
  76.         return 1;
  77.     case '}':
  78.     case ']':
  79.     case ')':
  80.     case '>':
  81.         return 2;
  82.     }
  83.     return 0;
  84. }
  86. int main()
  87. {
  88.     string tmp;
  89.     getline(cin,tmp);
  90.     mystack*stk=new mystack();
  91.     bool erred=false;
  92.     for(int i=0;i<tmp.size();i++){
  93.         char x=tmp.at(i);
  94.         if(x=='#'){
  95.             if(!stk->isEmpty()){
  96.                 cout<<"loss of right character ";
  97.                 while(!stk->isEmpty()){
  98.                     cout<<getPar(stk->getTop());
  99.                     stk->pop();
  100.                 }
  101.                 cout<<"."<<endl;
  102.                 //erred=true;
  103.                 return 0;
  104.             }
  105.             if(!erred){
  106.                 cout<<"right"<<endl;
  107.             }
  108.             return 0;
  109.         }
  110.         if(isKH(x)==1){
  111.             stk->push(x);
  112.             continue;
  113.         }
  114.         if(isKH(x)==2){
  115.             if(getPar(stk->getTop())==x){
  116.                 stk->pop();
  117.                 continue;
  118.             }else{
  119.                 cout<<"The "<<i+1<<" character '"<<x<<"' is wrong."<<endl;
  120.                 //erred=true;
  121.                 return 0;
  122.             }
  123.         }
  124.     }
  125.     //cout << "Hello world!" << endl;
  126.     return 0;
  127. }
